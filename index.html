            const isHappy = (str) => {
              if (str.length % 2 === 0) {
                let strLeft = 0;
                for (let i = 0; i < str.length / 2; i += 1) {
                  strLeft += Number(str[i]);
                  }
                let strRight = 0;
                for (let i = str.length - 1; i > (str.length - 1) / 2; i -= 1) {
                  strRight += Number(str[i]);
                }
                return strLeft === strRight;
              }
              return false;
            };
            export default isHappy;
